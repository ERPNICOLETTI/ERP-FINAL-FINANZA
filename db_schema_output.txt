DATABASE SCHEMA
===============

--- TABLE: bancos_movimientos ---
CREATE TABLE bancos_movimientos (
            id              INTEGER PRIMARY KEY AUTOINCREMENT,
            banco           TEXT,
            cuenta          TEXT,
            fecha           TEXT,
            descripcion     TEXT,
            tipo_movimiento TEXT,
            importe         REAL DEFAULT 0,
            saldo           REAL,
            hash_archivo    TEXT,
            metadata_cruda  TEXT DEFAULT '{}',
            created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
            UNIQUE(banco, cuenta, fecha, descripcion, tipo_movimiento, importe)
        )

Columns:
  id (INTEGER) PK
  banco (TEXT) 
  cuenta (TEXT) 
  fecha (TEXT) 
  descripcion (TEXT) 
  tipo_movimiento (TEXT) 
  importe (REAL) 
  saldo (REAL) 
  hash_archivo (TEXT) 
  metadata_cruda (TEXT) 
  created_at (TIMESTAMP)

--- TABLE: facturas ---

--- TABLE: liquidaciones_detalles ---
CREATE TABLE liquidaciones_detalles (
            id              INTEGER PRIMARY KEY AUTOINCREMENT,
            liquidacion_id  INTEGER,
            fecha           TEXT,
            descripcion     TEXT,
            monto_bruto     REAL DEFAULT 0,
            arancel         REAL DEFAULT 0,
            financiero      REAL DEFAULT 0,
            iva             REAL DEFAULT 0,
            retenciones     REAL DEFAULT 0,
            monto_neto      REAL DEFAULT 0,
            metadata_cruda  TEXT DEFAULT '{}',
            FOREIGN KEY(liquidacion_id) REFERENCES liquidaciones_tarjetas(id)
        )

Columns:
  id (INTEGER) PK
  liquidacion_id (INTEGER) 
  fecha (TEXT) 
  descripcion (TEXT) 
  monto_bruto (REAL) 
  arancel (REAL) 
  financiero (REAL) 
  iva (REAL) 
  retenciones (REAL) 
  monto_neto (REAL) 
  metadata_cruda (TEXT)

            metadata_cruda  TEXT DEFAULT '{}',
            created_at      TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        )

Columns:

  fuente (TEXT) 
  marca (TEXT) 
  tipo (TEXT) 
  fecha_liquidacion (TEXT) 
  periodo (TEXT) 
  establecimiento (TEXT)

  metadata_cruda (TEXT) 
  created_at (TIMESTAMP)

            fecha_pago      TEXT,
            lote            TEXT,
            cupon           TEXT,

        )

Columns:
  id (INTEGER) PK
  fuente (TEXT) 
  fecha_compra (TEXT) 
  fecha_pago (TEXT)

  marca (TEXT) 
  monto_bruto (REAL)

  hash_archivo (TEXT) 
  metadata_cruda (TEXT) 
  created_at (TIMESTAMP)
